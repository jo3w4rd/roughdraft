    Struct ParticleSystem.RotationOverLifetimeModule

    Script interface for the Rotation Over Lifetime module.

    Namespace: UnityEngine
    Syntax
    public struct RotationOverLifetimeModule
    Remarks

    SA: ParticleSystem, ParticleSystem.rotationOverLifetime.

    Properties

    enabled

    Enable/disable the Rotation Over Lifetime module.

    Declaration
    public bool enabled { get; set; }
    Property Value
    Type Description
    Boolean
    Remarks

    SA: ParticleSystem.rotationOverLifetime.

    Examples
    
using UnityEngine;
using System.Collections;

public class ExampleClass : MonoBehaviour
{
    private ParticleSystem ps;
    public float hSliderValue = 1.0f;

    void Start()
    {
        ps = GetComponent<ParticleSystem>();

        var rotationOverLifetime = ps.rotationOverLifetime;
        rotationOverLifetime.enabled = true;

        AnimationCurve curve = new AnimationCurve();
        curve.AddKey(0.0f, 0.0f);
        curve.AddKey(1.0f, 1.0f);

        rotationOverLifetime.z = new ParticleSystem.MinMaxCurve(Mathf.PI * 2.0f, curve);

        var psr = GetComponent<ParticleSystemRenderer>();
        psr.material = new Material(Shader.Find("Sprites/Default"));    // this material renders a square billboard, so we can see the rotation
    }

    void Update()
    {
        var main = ps.main;
        main.startLifetime = hSliderValue;
    }

    void OnGUI()
    {
        hSliderValue = GUI.HorizontalSlider(new Rect(25, 45, 100, 30), hSliderValue, 1.0f, 5.0f);
    }
}

    separateAxes

    Set the rotation over lifetime on each axis separately.

    Declaration
    public bool separateAxes { get; set; }
    Property Value
    Type Description
    Boolean
    Remarks
