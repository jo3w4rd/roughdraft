    fontSize

    The font size to use (for dynamic fonts).

    Declaration
    public int fontSize { get; set; }
    Property Value
    Type Description
    Int32
    Remarks

    If this is set to a non-zero value, the font size specified in the font importer is overriden with a custom size. This is only supported for fonts set to use dynamic font rendering. Other fonts will always use the default font size.

    Examples
    
using UnityEngine;

public class Example : MonoBehaviour
{
    void Start()
    {
        GetComponent<TextMesh>().fontSize = 12;
    }
}

    richText

    Enable HTML-style tags for Text Formatting Markup.

    Declaration
    public bool richText { get; set; }
    Property Value
    Type Description
    Boolean
    Remarks

    Supported tags are:

    <color="htmlcolor">colored text, where "htmlcolor" is a html color string, like "#ff0000" or "red".

    bold text

    italic text

    <size=20>sized text <material=1>render using custom material index

    <quad material=1 size=20 x=0.1 y=0.1 width=0.5 height=0.5/>, to render a single quad using the given material and UVs, used for embedding images in text.

    These are only supported for fonts set to use dynamic font rendering, except for the 'color', 'material' and 'quad' tags.

    Examples
    
using UnityEngine;

public class Example : MonoBehaviour
{
    void Start()
    {
        GetComponent<TextMesh>().richText = true;
    }
}
