    Class WindZone

    Wind Zones add realism to the trees you create by making them wave their branches and leaves as if blown by the wind.

    Remarks

    Note: This only works with trees created by the tree creator or imported from SpeedTree Modeler.

    Properties

    mode

    Defines the type of wind zone to be used (Spherical or Directional).

    Declaration
    public WindZoneMode mode { get; set; }
    Property Value
    Type Description
    WindZoneMode
    Examples
    
// Creates a Directional Wind Zone.
using UnityEngine;

public class ExampleScript : MonoBehaviour
{
    void Start()
    {
        var wind = gameObject.AddComponent<WindZone>();
        wind.mode = WindZoneMode.Directional;
    }
}

    windMain

    The primary wind force.

    Declaration
    public float windMain { get; set; }
    Property Value
    Type Description
    Single
    Remarks

    It produces a softly changing wind Pressure.

    Examples
    
// Creates a wind zone with the effect of a helicopter passing by
// Just place this into an empty game object and move it over a tree
using UnityEngine;

public class ExampleScript : MonoBehaviour
{
    void Start()
    {
        var wind = gameObject.AddComponent<WindZone>();
        wind.mode = WindZoneMode.Spherical;
        wind.radius = 10.0f;
        wind.windMain = 3.0f;
        wind.windTurbulence = 0.5f;
        wind.windPulseMagnitude = 2.0f;
        wind.windPulseFrequency = 0.01f;
    }
}
