    Examples
    
using UnityEngine;

public class Example : MonoBehaviour
{
    // Print the format of a given cubemap.
    public Cubemap cubeMap;

    void Start()
    {
        if (cubeMap != null)
        {
            Debug.Log(cubeMap.format);
        }
        else
        {
            Debug.Log("No cubemap was assigned, please assing one on the inspector.");
        }
    }
}
