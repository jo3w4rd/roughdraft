    Class Logger

    Initializes a new instance of the Logger.

    Inheritance
    Object
    Logger
    Namespace: UnityEngine
    Syntax
    public class Logger : ILogger, ILogHandler
    Remarks

    Create a new instance or use default Debug.unityLogger. SA: ILogger, ILogHandler.

    Examples
    
using UnityEngine;
using System.Collections;
using System.IO;
using System;

public class MyLogHandler : ILogHandler
{
    public void LogFormat(LogType logType, UnityEngine.Object context, string format, params object[] args)
    {
        Debug.unityLogger.logHandler.LogFormat(logType, context, format, args);
    }

    public void LogException(Exception exception, UnityEngine.Object context)
    {
        Debug.unityLogger.LogException(exception, context);
    }
}

public class MyGameClass : MonoBehaviour
{
    private static string kTAG = "MyGameTag";
    private Logger myLogger;

    void Start()
    {
        myLogger = new Logger(new MyLogHandler());

        myLogger.Log(kTAG, "MyGameClass Start.");
    }
}

    filterLogType

    To selective enable debug log message.

    Declaration
    public LogType filterLogType { get; set; }
    Property Value
    Type Description
    LogType
    Implements
    ILogger.filterLogType
    Remarks

    By settings filterLogType to

    [LogType.Log](xref:UnityEngine.LogType.Log)  (default setting) will display all log messages.

[LogType.Warning](xref:UnityEngine.LogType.Warning) will display Warning, Assert, Error and Exception log messages.

[LogType.Assert](xref:UnityEngine.LogType.Assert) will display Assert, Error and Exception log messages.

[LogType.Error](xref:UnityEngine.LogType.Error) will display Error and Exception log messages.

[LogType.Exception](xref:UnityEngine.LogType.Exception) will display Exception log messages.
    Examples
    
using UnityEngine;
using System.Collections;

public class MyGameClass : MonoBehaviour
{
    private static ILogger logger = Debug.unityLogger;
    private static string kTAG = "MyGameTag";

    void Start()
    {
        if (Debug.isDebugBuild)
            logger.filterLogType = LogType.Log;
        else
            logger.filterLogType = LogType.Warning;

        logger.Log(kTAG, "This log will be displayed only in debug build");
        logger.LogError(kTAG, "This log will be displayed in debug and release build");
    }
}
