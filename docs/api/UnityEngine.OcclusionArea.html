    Examples
    
using UnityEngine;

[RequireComponent(typeof(OcclusionArea))]
public class Example : MonoBehaviour
{
    // Sets the center of the occlusion area to the center of the transform
    void Start()
    {
        transform.GetComponent<OcclusionArea>().center = Vector3.zero;
    }
